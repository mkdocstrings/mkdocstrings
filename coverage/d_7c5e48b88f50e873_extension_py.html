1"""This module holds the code of the Markdown extension responsible for matching "autodoc" instructions. 

2 

3The extension is composed of a Markdown [block processor](https://python-markdown.github.io/extensions/api/#blockparser) 

4that matches indented blocks starting with a line like '::: identifier'. 

5 

6For each of these blocks, it uses a [handler][mkdocstrings.handlers.base.BaseHandler] to collect documentation about 

7the given identifier and render it with Jinja templates. 

8 

9Both the collection and rendering process can be configured by adding YAML configuration under the "autodoc" 

10instruction: 

11 

12```yaml 

13::: some.identifier 

14 handler: python 

15 selection: 

16 option1: value1 

17 option2: 

18 - value2a 

19 - value2b 

20 rendering: 

21 option_x: etc 

22``` 

23"""

51class AutoDocProcessor(BlockProcessor): 

52 """Our "autodoc" Markdown block processor. 

53 

54 It has a [`test` method][mkdocstrings.extension.AutoDocProcessor.test] that tells if a block matches a criterion, 

55 and a [`run` method][mkdocstrings.extension.AutoDocProcessor.run] that processes it. 

56 

57 It also has utility methods allowing to get handlers and their configuration easily, useful when processing 

58 a matched block. 

59 """

95 def run(self, parent: Element, blocks: MutableSequence[str]) -> None: 

96 """Run code on the matched blocks. 

97 

98 The identifier and configuration lines are retrieved from a matched block 

99 and used to collect and render an object. 

100 

101 Arguments: 

102 parent: The parent element in the XML tree. 

103 blocks: The rest of the blocks to be processed. 

104 """ 

105 block = blocks.pop(0) 

106 match = self.regex.search(block) 

107 

108 if match: 108 ↛ 114line 108 didn't jump to line 114, because the condition on line 108 was never false

109 if match.start() > 0: 

110 self.parser.parseBlocks(parent, [block[: match.start()]]) 

111 # removes the first line 

112 block = block[match.end() :] 

113 

114 block, the_rest = self.detab(block) 

115 

116 if match: 116 ↛ 145line 116 didn't jump to line 145, because the condition on line 116 was never false

117 identifier = match["name"] 

118 heading_level = match["heading"].count("#") 

119 log.debug(f"Matched '::: {identifier}'") 

120 

121 html, handler, data = self._process_block(identifier, block, heading_level) 

122 el = Element("div", {"class": "mkdocstrings"}) 

123 # The final HTML is inserted as opaque to subsequent processing, and only revealed at the end. 

124 el.text = self.md.htmlStash.store(html) 

125 # So we need to duplicate the headings directly (and delete later), just so 'toc' can pick them up. 

126 headings = handler.get_headings() 

127 el.extend(headings) 

128 

129 page = self._autorefs.current_page 

130 if page: 

131 for heading in headings: 

132 anchor = heading.attrib["id"] # noqa: WPS440 

133 self._autorefs.register_anchor(page, anchor) # noqa: WPS441 

134 

135 if "data-role" in heading.attrib: 135 ↛ 131line 135 didn't jump to line 131, because the condition on line 135 was never false

137 name=anchor, # noqa: WPS441 

138 domain=handler.domain, 

139 role=heading.attrib["data-role"], 

140 uri=f"{page}#{anchor}", # noqa: WPS441 

141 ) 

142 

143 parent.append(el) 

144 

145 if the_rest: 145 ↛ 149line 145 didn't jump to line 149, because the condition on line 145 was never true

146 # This block contained unindented line(s) after the first indented 

147 # line. Insert these lines as the first block of the master blocks 

148 # list for future processing. 

149 blocks.insert(0, the_rest)

151 def _process_block( 

152 self, 

153 identifier: str, 

154 yaml_block: str, 

155 heading_level: int = 0, 

156 ) -> Tuple[str, BaseHandler, CollectorItem]: 

157 """Process an autodoc block. 

158 

159 Arguments: 

160 identifier: The identifier of the object to collect and render. 

161 yaml_block: The YAML configuration. 

162 heading_level: Suggested level of the the heading to insert (0 to ignore). 

163 

164 Raises: 

165 PluginError: When something wrong happened during collection. 

166 TemplateNotFound: When a template used for rendering could not be found. 

167 

168 Returns: 

169 Rendered HTML, the handler that was used, and the collected item. 

170 """ 

171 config = yaml.safe_load(yaml_block) or {} 

172 handler_name = self._handlers.get_handler_name(config) 

173 

174 log.debug(f"Using handler '{handler_name}'") 

175 handler_config = self._handlers.get_handler_config(handler_name) 

176 handler = self._handlers.get_handler(handler_name, handler_config) 

177 

178 global_options = handler_config.get("options", {}) 

179 local_options = config.get("options", {}) 

180 deprecated_global_options = ChainMap(handler_config.get("selection", {}), handler_config.get("rendering", {})) 

181 deprecated_local_options = ChainMap(config.get("selection", {}), config.get("rendering", {})) 

182 

183 options = ChainMap(local_options, deprecated_local_options, global_options, deprecated_global_options) 

184 

185 if deprecated_global_options or deprecated_local_options: 

186 self._warn_about_options_key() 

187 

188 if heading_level: 188 ↛ 189line 188 didn't jump to line 189, because the condition on line 188 was never true

189 options = ChainMap(options, {"heading_level": heading_level}) # like setdefault 

190 

191 log.debug("Collecting data") 

192 try: 

193 data: CollectorItem = handler.collect(identifier, options) 

194 except CollectionError as exception: 

195 log.error(str(exception)) 

196 if PluginError is SystemExit: # When MkDocs 1.2 is sufficiently common, this can be dropped. 

197 log.error(f"Error reading page '{self._autorefs.current_page}':") 

198 raise PluginError(f"Could not collect '{identifier}'") from exception 

199 

200 if not self._updated_env: 

201 log.debug("Updating renderer's env") 

202 handler._update_env(self.md, self._config) # noqa: WPS437 (protected member OK) 

203 self._updated_env = True 

204 

205 log.debug("Rendering templates") 

206 try: 

207 rendered = handler.render(data, options) 

208 except TemplateNotFound as exc: 

209 theme_name = self._config["theme_name"] 

210 log.error( 

211 f"Template '{exc.name}' not found for '{handler_name}' handler and theme '{theme_name}'.", 

212 ) 

213 raise 

214 

215 return rendered, handler, data

223class _PostProcessor(Treeprocessor): 

224 def run(self, root: Element): 

225 carry_text = "" 

226 for el in reversed(root): # Reversed mainly for the ability to mutate during iteration. 

227 if el.tag == "div" and el.get("class") == "mkdocstrings": 

228 # Delete the duplicated headings along with their container, but keep the text (i.e. the actual HTML). 

229 carry_text = (el.text or "") + carry_text 

230 root.remove(el) 

231 elif carry_text: 

232 el.tail = (el.tail or "") + carry_text 

233 carry_text = "" 

234 if carry_text: 

235 root.text = (root.text or "") + carry_text
