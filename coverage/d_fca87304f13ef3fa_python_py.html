120 

121 default_config: dict = {"filters": ["!^_[^_]"]} 

122 """The default selection options. 

123 

124 Option | Type | Description | Default 

125 ------ | ---- | ----------- | ------- 

126 **`filters`** | `List[str]` | Filter members with regular expressions. | `[ "!^_[^_]" ]` 

127 **`members`** | `Union[bool, List[str]]` | Explicitly select the object members. | *`pytkdocs` default: `True`* 

128 

129 If `members` is a list of names, filters are applied only on the members children (not the members themselves). 

130 If `members` is `False`, none are selected. 

131 If `members` is `True` or an empty list, filters are applied on all members and their children. 

132 

133 Members affect only the first layer of objects, while filters affect the whole object-tree recursively. 

134 

135 Every filters is run against every object name. An object can be un-selected by a filter and re-selected by the 

136 next one: 

137 

138 - `"!^_"`: exclude all objects starting with an underscore 

139 - `"^__"`: but select all objects starting with **two** underscores 

140 

141 Obviously one could use a single filter instead: `"!^_[^_]"`, which is the default. 

142 """

144 def __init__(self, setup_commands: Optional[List[str]] = None) -> None: 

145 """Initialize the object. 

146 

147 When instantiating a Python collector, we open a subprocess in the background with `subprocess.Popen`. 

148 It will allow us to feed input to and read output from this subprocess, keeping it alive during 

149 the whole documentation generation. Spawning a new Python subprocess for each "autodoc" instruction would be 

150 too resource intensive, and would slow down `mkdocstrings` a lot. 

151 

152 Arguments: 

153 setup_commands: A list of python commands as strings to be executed in the subprocess before `pytkdocs`. 

154 """ 

155 log.debug("Opening 'pytkdocs' subprocess") 

156 env = os.environ.copy() 

157 env["PYTHONUNBUFFERED"] = "1" 

158 

159 if setup_commands: 

160 # prevent the Python interpreter or the setup commands 

161 # from writing to stdout as it would break pytkdocs output 

162 commands = [ 

163 "import sys", 

164 "from io import StringIO", 

165 "from pytkdocs.cli import main as pytkdocs", 

166 "sys.stdout = StringIO()", # redirect stdout to memory buffer 

167 *setup_commands, 

168 "sys.stdout.flush()", 

169 "sys.stdout = sys.__stdout__", # restore stdout 

170 "pytkdocs(['--line-by-line'])", 

171 ] 

172 cmd = [sys.executable, "-c", "; ".join(commands)] 

173 else: 

174 cmd = [sys.executable, "-m", "pytkdocs", "--line-by-line"] 

175 

176 self.process = Popen( # noqa: S603,S607 (we trust the input, and we don't want to use the absolute path) 

177 cmd,

185 def collect(self, identifier: str, config: dict) -> CollectorItem: 

186 """Collect the documentation tree given an identifier and selection options. 

187 

188 In this method, we feed one line of JSON to the standard input of the subprocess that was opened 

189 during instantiation of the collector. Then we read one line of JSON on its standard output. 

190 

191 We load back the JSON text into a Python dictionary. 

192 If there is a decoding error, we log it as error and raise a CollectionError. 

193 

194 If the dictionary contains an `error` key, we log it as error (with the optional `traceback` value), 

195 and raise a CollectionError. 

196 

197 If the dictionary values for keys `loading_errors` and `parsing_errors` are not empty, 

198 we log them as warnings. 

199 

200 Then we pick up the only object within the `objects` list (there's always only one, because we collect 

201 them one by one), rebuild it's categories lists 

202 (see [`rebuild_category_lists()`][mkdocstrings.handlers.python.rebuild_category_lists]), 

203 and return it. 

204 

205 Arguments: 

206 identifier: The dotted-path of a Python object available in the Python path. 

207 config: Selection options, used to alter the data collection done by `pytkdocs`. 

208 

209 Raises: 

210 CollectionError: When there was a problem collecting the object documentation. 

211 

212 Returns: 

213 The collected object-tree. 

214 """ 

215 final_config = ChainMap(config, self.default_config) 

216 

217 log.debug("Preparing input") 

218 json_input = json.dumps({"objects": [{"path": identifier, **final_config}]}) 

219 

220 log.debug("Writing to process' stdin") 

221 self.process.stdin.write(json_input + "\n") # type: ignore 

222 self.process.stdin.flush() # type: ignore 

223 

224 log.debug("Reading process' stdout") 

225 stdout = self.process.stdout.readline() # type: ignore 

226 

227 log.debug("Loading JSON output as Python object") 

228 try: 

229 result = json.loads(stdout) 

230 except json.decoder.JSONDecodeError as exception: 

231 error = "\n".join(("Error while loading JSON:", stdout, traceback.format_exc())) 

232 raise CollectionError(error) from exception 

233 

234 error = result.get("error") 

235 if error: 235 ↛ 236line 235 didn't jump to line 236, because the condition on line 235 was never true

236 if "traceback" in result: 

237 error += f"\n{result['traceback']}" 

238 raise CollectionError(error) 

239 

240 for loading_error in result["loading_errors"]: 240 ↛ 241line 240 didn't jump to line 241, because the loop on line 240 never started

241 log.warning(loading_error) 

242 

243 for errors in result["parsing_errors"].values(): 243 ↛ 244line 243 didn't jump to line 244, because the loop on line 243 never started

244 for parsing_error in errors: 

245 log.warning(parsing_error) 

246 

247 # We always collect only one object at a time 

248 result = result["objects"][0] 

249

252 

253 return result

320def rebuild_category_lists(obj: dict) -> None: 

321 """Recursively rebuild the category lists of a collected object. 

322 

323 Since `pytkdocs` dumps JSON on standard output, it must serialize the object-tree and flatten it to reduce data 

324 duplication and avoid cycle-references. Indeed, each node of the object-tree has a `children` list, containing 

325 all children, and another list for each category of children: `attributes`, `classes`, `functions`, `methods` 

326 and `modules`. It replaces the values in category lists with only the paths of the objects. 

327 

328 Here, we reconstruct these category lists by picking objects in the `children` list using their path. 

329 

330 For each object, we recurse on every one of its children. 

331 

332 Arguments: 

333 obj: The collected object, loaded back from JSON into a Python dictionary. 

334 """ 

335 for category in ("attributes", "classes", "functions", "methods", "modules"): 

336 obj[category] = [obj["children"][path] for path in obj[category]] 

337 obj["children"] = [child for _, child in obj["children"].items()] 

338 for child in obj["children"]: 

339 rebuild_category_lists(child)
